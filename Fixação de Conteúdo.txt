   VALIDADE DATE,

• Mostrar todos os dados dentro de um intervalo de data

SELECT * FROM PRODUTOS WHERE VALIDADE > TO_DATE('30/12/2026','DD/MM/YYYY');

• Escolha apenas dois ou três colunas para que seja mostrado dentro de um intervalo de preço em ordem
numérica

SELECT NOME,PRECO,ESTADO_PRODUTOR FROM PRODUTOS ORDER BY PRECO ASC;

• Mostrar somente os produtos que comece com uma letra específica

SELECT * FROM PRODUTOS WHERE NOME LIKE 'A%';

• Mostrar os dados de um mês em específico (pesquisar na internet)

SELECT * FROM PRODUTOS WHERE TO_CHAR (VALIDADE,'MM') =03;

• Crie 3 condições para que algum ou alguns dados sejam alterados

UPDATE PRODUTOS SET PRECO=2.99 WHERE PESO_LIQ<500 AND QUANTIDADE<5 AND PRECO=1.99;

• Crie 3 condições para que algum ou alguns dados sejam excluídos

DELETE FROM PRODUTOS WHERE  PESO_LIQ<1000 AND PRECO<3 AND QUANTIDADE=1;
